## 作业一 EP09

```c
#include <stdio.h>
#include <math.h>

int gcd(int m, int n) {
    return (n == 0 ? m : gcd(n, m % n));
}

int main() {
    int n = 0, a, b, c;
    while (n++ < 16) { 
        int m = n;
        while (m++ < 16) { // a, b 一奇一偶 n, m互质 m > n
            if (gcd(m, n) != 1) continue;
            a = 2 * n * m;
            b = m * m - n * n;
            c = m * m + n * n;
            if (1000 % (a + b + c) == 0)
                printf("%d\n", a * b * c * (int)pow(1000 / (a + b + c), 3));
        }
    }
    return 0;
}
```



## 作业一 EP20

```c
#include <stdio.h>

int num[300] = {1, 1};

int main() {
    int sum = 0;
    for (int i = 1; i <= 100; ++i) {
        for (int j = 1; j <= num[0]; j++) num[j] *= i;
        for (int j = 1; j <= num[0]; j++) { // 大数
            if (num[j] < 10) continue;
            num[j + 1] += num[j] / 10;
            num[j] %= 10;
            num[0] += (j == num[0]);
        }
    }
    for (int i = num[0]; i > 0; i--) sum += num[i];
    printf("%d\n", sum);
    return 0;
}
```



## 作业一 EP22

```c
#include <iostream>
#include <algorithm>
#include <cstdio>
#include "EP22.h"

using namespace std;

int main() {
    int i = 0, sum = 0, ans = 0;
    while (Name[i++].length() != 0); // 计算位数
    sort(Name, Name + i - 1); // 字典序排序
    for (i = 0; Name[i].length() != 0; i++) {
        for (int j = 0; j < Name[i].length(); j++) {
            ans += (Name[i][j] - 'A' + 1);
        }
        sum += (i + 1) * ans;
        ans = 0;
    }
    printf("%d\n", sum);
}
```
